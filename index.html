Codeforces 1499F Diameter Cuts 题解

发表于 2023-09-23 分类于 题解

思路

设 fu,if_{u, i}fu,i​ 表示 uuu 的子树内，以 uuu 开始的链最长的长度为 iii 的方案数．

转移考虑合并一棵子树 vvv，枚举 u,vu, vu,v 中最长链的长度 i,ji, ji,j，有

  1. i+j≤ki + j \le ki+j≤k 时，此时可以选择不断开边 (u,v)(u, v)(u,v)，也可选择断开 (u,v)(u, v)(u,v)，有转移 fu,max⁡{i,j+1}←2×fu,ifv,jf_{u, \max\{i, j + 1\}} \leftarrow 2 \times f_{u, i} f_{v, j}fu,max{i,j+1}​←2×fu,i​fv,j​．
  2. i+j>ki + j > ki+j>k 时，此时只能选择断开 (u,v)(u, v)(u,v)，有转移 fu,max⁡{i,j+1}←fu,ifv,jf_{u, \max\{i, j + 1\}} \leftarrow f_{u, i} f_{v, j}fu,max{i,j+1}​←fu,i​fv,j​．

看似是 Θ(nk2)\Theta(nk^2)Θ(nk2) 的，但事实上若枚举上界定为子树深度，那么可以分析出时间复杂度为 O(nk)O(nk)O(nk)．分析参考 这篇博客．

代码

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#define debug(...) fprintf(stderr, __VA_ARGS__)

inline int rd() {
	int x = 0, f = 1, c = getchar();
	while (((c - '0') | ('9' - c)) < 0)
		f = c != '-', c = getchar();
	while (((c - '0') | ('9' - c)) > 0)
		x = x * 10 + c - '0', c = getchar();
	return f ? x : -x;
}

using i64 = long long;
const i64 P = 998244353;

const int N = 5e3;

int n, k; std::vector<int> g[N + 5];
i64 f[N + 5][N + 5]; int dep[N + 5];
void dfs(int u, int fa) {
	for(int v : g[u]) if(v != fa) dfs(v, u);

	static i64 h[N + 5];
	f[u][1] = 1, dep[u] = 1;
	for(int v : g[u]) {
		if(v == fa) continue;
		memset(h, 0, sizeof(h));
		for(int i = 1; i <= dep[u]; i++) {
			for(int j = 1; j <= dep[v]; j++) {
				if(i + j <= k) (h[std::max(i, j + 1)] += f[u][i] * f[v][j] % P) %= P;
				(h[i] += f[u][i] * f[v][j] % P) %= P;
			}
		}
		dep[u] = std::max(dep[u], dep[v] + 1);
		memcpy(f[u], h, sizeof(f[u]));
	}
}

int main() {
	n = rd(), k = rd() + 1;
	for(int i = 1; i < n; i++) {
		int u = rd(), v = rd();
		g[u].push_back(v), g[v].push_back(u);
	}

	dfs(1, 0);
	i64 ans = 0;
	for(int i = 1; i <= k; i++) (ans += f[1][i]) %= P;
	printf("%lld\n", ans);
	return 0;
}

Codeforces 1120C Compress String 题解

发表于 2023-09-23 分类于 题解

思路

设 fif_ifi​ 表示考虑了前 iii 个字符的最小代价，有转移

fi←fi−1+afi←fi−k+b\begin{aligned} f_i &\leftarrow f_{i - 1} + a \\ f_i &\leftarrow f_{i - k} + b \end{aligned} fi​fi​​←fi−1​+a←fi−k​+b​

其中 kkk 满足 s[i−k+1:i]s[i - k + 1 : i]s[i−k+1:i] 是 s[:i−k]s[: i - k]s[:i−k] 的子串．

这样决策点很难找，不妨直接枚举子串的结束位置 jjj，设 lcsi,j\mathrm{lcs}_{i, j}lcsi,j​ 表示前缀 s[:i]s[: i]s[:i] 和 s[:j]s[: j]s[:j] 的最长公共后缀，有转移 fi←fmax⁡{i−lcsi,j,j}f_i \leftarrow f_{\max\{i - \mathrm{lcs}_{i, j}, j\}}fi​←fmax{i−lcsi,j​,j}​．这样就能做到 Θ(n)\Theta(n)Θ(n) 转移了．

代码

#include <algorithm>
#include <cstdio>
#include <cstring>
#define debug(...) fprintf(stderr, __VA_ARGS__)

inline int rd() {
	int x = 0, f = 1, c = getchar();
	while (((c - '0') | ('9' - c)) < 0)
		f = c != '-', c = getchar();
	while (((c - '0') | ('9' - c)) > 0)
		x = x * 10 + c - '0', c = getchar();
	return f ? x : -x;
}

const int N = 5e3;

int n, a, b;
char s[N + 5];

int lcs[N + 5][N + 5];
int f[N + 5];

int main() {
	n = rd(), a = rd(), b = rd(), scanf("%s", s + 1);
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= n; j++) {
			if (s[i] == s[j]) lcs[i][j] = lcs[i - 1][j - 1] + 1;
		}
	}
	memset(f, 63, sizeof(f)), f[0] = 0;
	for (int i = 1; i <= n; i++) {
		f[i] = f[i - 1] + a;
		for (int j = 1; j < i; j++) {
			f[i] = std::min(f[i], f[std::max(i - lcs[i][j], j)] + b);
		}
	}
	printf("%d\n", f[n]);
	return 0;
}

Codeforces 1228E Another Filling the Grid 题解

发表于 2023-09-23 分类于 题解

思路

设 gi,jg_{i, j}gi,j​ 表示恰有 iii 行 jjj 列最小值为 111 的方案，fi,jf_{i, j}fi,j​ 表示钦定 iii 行 jjj 列，使得剩下的行列最小值不为 111 的方案数，有

fn,n=∑i=0n∑j=0n(ni)(nj)gi,j  ⟺  gn,n=∑i=0n∑j=0n(ni)(nj)(−1)n−i+n−jfi,jf_{n, n} = \sum_{i = 0}^n \sum_{j = 0}^n \binom{n}{i} \binom{n}{j} g_{i, j} \iff g_{n, n} = \sum_{i = 0}^n \sum_{j = 0}^n \binom{n}{i} \binom{n}{j} (-1)^{n - i + n - j} f_{i, j} fn,n​=i=0∑n​j=0∑n​(in​)(jn​)gi,j​⟺gn,n​=i=0∑n​j=0∑n​(in​)(jn​)(−1)n−i+n−jfi,j​

其中 fi,jf_{i, j}fi,j​ 显然为 (k−1)n2−(n−i)(n−j)×k(n−i)(n−j)(k - 1)^{n^2 - (n - i)(n - j)} \times k^{(n - i)(n - j)}(k−1)n2−(n−i)(n−j)×k(n−i)(n−j)．

可以直接计算．

代码

#include <cstdio>
#define debug(...) fprintf(stderr, __VA_ARGS__)

inline int rd() {
	int x = 0, f = 1, c = getchar();
	while (((c - '0') | ('9' - c)) < 0)
		f = c != '-', c = getchar();
	while (((c - '0') | ('9' - c)) > 0)
		x = x * 10 + c - '0', c = getchar();
	return f ? x : -x;
}

using i64 = long long;
const i64 P = 1e9 + 7;

i64 fpow(i64 b, i64 p) {
	i64 res = 1;
	for (; p; b = b * b % P, p >>= 1) {
		if (p & 1) res = res * b % P;
	}
	return res;
}

const int N = 250;

int n;
i64 k;

i64 C[N + 5][N + 5];
void pre() {
	for (int i = 0; i <= N; i++) C[i][0] = 1;
	for (int i = 1; i <= N; i++) {
		for (int j = 1; j <= N; j++) {
			C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % P;
		}
	}
}

int main() {
	pre();

	n = rd(), k = rd();

	i64 ans = 0;
	for (int i = 0; i <= n; i++) {
		for (int j = 0; j <= n; j++) {
			i64 coef = C[n][i] * C[n][j] % P * fpow(k, (n - i) * (n - j)) % P * fpow(k - 1, n * n - (n - i) * (n - j)) % P;
			if ((n * 2 - i - j) & 1) (ans += P - coef) %= P;
			else (ans += coef) %= P;
		}
	}
	printf("%lld\n", ans);
	return 0;
}

Codeforces 1515E Phoenix and Computers 题解

发表于 2023-09-22 分类于 题解

思路

设 fi,jf_{i, j}fi,j​ 表示已经打开了 iii 台电脑，已经打开的电脑构成了 jjj 个连续段的方案数．

考虑对连续段操作：

  1. 延长连续段．可以选择在某个连续段两边插入，或间隔一个插入，第二种操作会导致中间的那个未被打开的电脑被打开，有 2j×fi,j→fi+1,j2j \times f_{i, j} \to f_{i + 1, j}2j×fi,j​→fi+1,j​，2j×fi,j→fi+2,j2j \times f_{i, j} \to f_{i + 2, j}2j×fi,j​→fi+2,j​，

  2. 合并连续段．间隔两个格子的连续段可以通过任意打开一台电脑来合并，有 2(j−1)×fi,j→fi+2,j−12(j - 1) \times f_{i, j} \to f_{i + 2, j - 1}2(j−1)×fi,j​→fi+2,j−1​；间隔三个格子的连续段可以通过打开中间的一台电脑合并，有 (j−1)×fi,j→i+3,j−1(j - 1) \times f_{i, j} \to {i + 3, j - 1}(j−1)×fi,j​→i+3,j−1．

  3. 新建连续段．(j+1)×fi,j→fi+1,j+1(j + 1) \times f_{i, j} \to f_{i + 1, j + 1}(j+1)×fi,j​→fi+1,j+1​．

代码

#include <cstdio>
#define debug(...) fprintf(stderr, __VA_ARGS__)

inline int rd() {
	int x = 0, f = 1, c = getchar();
	while (((c - '0') | ('9' - c)) < 0)
		f = c != '-', c = getchar();
	while (((c - '0') | ('9' - c)) > 0)
		x = x * 10 + c - '0', c = getchar();
	return f ? x : -x;
}

using i64 = long long;

const int N = 400;

int n;
i64 P, f[N + 5][N + 5];

int main() {
	n = rd(), P = rd();
	f[0][0] = 1;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j <= i; j++) {
			if (i + 1 <= n) (f[i + 1][j] += 2 * j * f[i][j] % P) %= P;
			if (i + 2 <= n) (f[i + 2][j] += 2 * j * f[i][j] % P) %= P;
			if (0 < j - 1 && i + 2 <= n) (f[i + 2][j - 1] += 2 * (j - 1) * f[i][j] % P) %= P;
			if (0 < j - 1 && i + 3 <= n) (f[i + 3][j - 1] += (j - 1) * f[i][j] % P) %= P;
			(f[i + 1][j + 1] += f[i][j] * (j + 1) % P) %= P;
		}
	}
	printf("%lld\n", f[n][1]);
	return 0;
}
