代码

#include <cstdio>

inline int rd() {
	int x = 0, f = 1, c = getchar();
	while (((c - '0') | ('9' - c)) < 0)
		f = c != '-', c = getchar();
	while (((c - '0') | ('9' - c)) > 0)
		x = x * 10 + c - '0', c = getchar();
	return f ? x : -x;
}

using i64 = long long;

const int N = 5e3;
const i64 P = 998244353;
i64 fpow(i64 b, i64 p) {
	i64 res = 1;
	for (; p; b = b * b % P, p >>= 1) {
		if (p & 1) res = res * b % P;
	}
	return res;
}

int n, m, k;
i64 S[N + 5][N + 5];

int main() {
	n = rd(), m = rd(), k = rd();

	S[0][0] = 1;
	for (int i = 1; i <= k; i++) {
		for (int j = 1; j <= k; j++) {
			S[i][j] = (S[i - 1][j - 1] + j * S[i - 1][j] % P) % P;
		}
	}

	i64 p = fpow(m, P - 2);
	i64 ans = 0;
	for (int i = 0; i <= k; i++) {
		i64 nj = 1;
		for (i64 j = n; j >= n - i + 1; j--) (nj *= j) %= P;
		(ans += S[k][i] * nj % P * fpow(p, i) % P) %= P;
	}

	printf("%lld\n", ans);
	return 0;
}
