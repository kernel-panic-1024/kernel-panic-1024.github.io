思路

考察一个区间 [l,r][l, r][l,r] 在何时可以被划分出并保留 [l,k][l, k][l,k]．

记 sl,r=⨁i=lrais_{l, r} = \bigoplus\limits_{i = l}^r a_isl,r​=i=l⨁r​ai​．对于一个划分点 kkk，记 x=sl,k,y=sk+1,rx = s_{l, k}, y = s_{k + 1, r}x=sl,k​,y=sk+1,r​．

  1. 若 sl,r=0s_{l, r} = 0sl,r​=0，那么对于任意的划分点 kkk，一定有 x=yx = yx=y，也就是说可以随便划分．
  2. 若 sl,r≠0s_{l, r} \not= 0sl,r​=0 时，考虑 sl,rs_{l, r}sl,r​ 的最高位 ppp，对于一个划分点 kkk，xxx 和 yyy 的第 ppp 位一定不同，且高于 ppp 的位数一定相同．也就是说，第 ppp 位决定了 xxx 和 yyy 的大小关系．

那么我们按照区间长度降序遍历所有区间，记 sti\mathrm{st}_isti​ 表示考察 iii 开始的合法区间，每个合法区间的异或和的最高位在 sti\mathrm{st}_isti​ 中的对应位都是 111．也就是说，对于 sti\mathrm{st}_isti​ 中的每一位 ppp，若该位上为 111，那么存在一个合法区间 [i,x][i, x][i,x]，使得 si,xs_{i, x}si,x​ 的最高位为 ppp．

当我们枚举一个区间 [l,k][l, k][l,k] 时，我们只需判断 sl,kand⁡stls_{l, k} \operatorname{and} \mathrm{st}_lsl,k​andstl​ 是否有值，若有值则 [l,k][l, k][l,k] 可以被某个区间划分出来．

#include <cstdio>
#define debug(...) fprintf(stderr, __VA_ARGS__)

#define int long long

inline int rd() {
	int x = 0, f = 1, c = getchar();
	while (((c - '0') | ('9' - c)) < 0)
		f = c != '-', c = getchar();
	while (((c - '0') | ('9' - c)) > 0)
		x = x * 10 + c - '0', c = getchar();
	return f ? x : -x;
}


const int N = 1e4;
const int U = (1ll << 61) - 1;

int n, a[N + 5], s[N + 5];

int stb[N + 5], edb[N + 5];

int highbit(int x) {
	if(!x) return U;
	return 1ll << (63 - __builtin_clzll(x));
}

void solve() {
	n = rd(); for(int i = 1; i <= n; i++) a[i] = rd();
	if(n == 1) return puts("1"), void();
	for(int i = 1; i <= n; i++) s[i] = s[i - 1] ^ a[i];
	stb[1] = edb[n] = highbit(s[n]);
	for(int len = n; len >= 1; len--) {
		for(int l = 1; l + len - 1 <= n; l++) {
			int r = l + len - 1;
			int sum = s[r] ^ s[l - 1];
			int msk = (stb[l] & sum) | (edb[r] & sum);
			int flg = 0;
			if(msk || stb[l] == U || edb[r] == U) {
				stb[l] |= highbit(sum), edb[r] |= highbit(sum);
				flg = 1;
			}
			if(len == 1) printf("%lld", flg);
		}
	}
	puts("");

	for(int i = 1; i <= n; i++) stb[i] = edb[i] = 0;
}

signed main() {
	int T = rd();
	while(T--) solve();
	return 0;
}
